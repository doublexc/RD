ตรวจสอบท้ายลิ้งค์ต้องเป็น ?id=name

(async function() { const urlParams = new URLSearchParams(window.location.search); const id = urlParams.get('id'); if (!id) return; let extraParams = ""; urlParams.forEach((value, key) => { if (key !== 'id') { extraParams += `&${key}=${value}`; } }); try { // ใส่ลิงก์ Google Sheets CSV ของคุณตรงนี้ const sheetUrl = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vQkTor2BGB5rV9i4fPjOKu39ocu_7e_fFXILbWFRJuxYeas4bsIHZAIGqoMyR6vLjf2Ei7dQhnMHYuR/pub?gid=461556905&single=true&output=csv'; const response = await fetch(sheetUrl); const data = await response.text(); const rows = data.split('\n'); for (let i = 1; i < rows.length; i++) { const cols = rows[i].split(','); if (cols[0] && cols[0].trim() === id.trim()) { let targetUrl = cols[1].trim(); // เช็คว่าลิงก์ปลายทางมี ? หรือยัง const separator = targetUrl.includes('?') ? '&' : '?'; // รวมร่างลิงก์ปลายทาง กับ Parameter ที่ส่งมาเพิ่ม (เช่น gift=123456) const finalUrl = extraParams ? `${targetUrl}${separator}${extraParams.substring(1)}` : targetUrl; window.location.replace(finalUrl); return; } } } catch (e) { console.error(e); } })(); })();
